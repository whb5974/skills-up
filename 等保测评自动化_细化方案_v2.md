## 一、方案概述

### 1.1 建设目标

构建自动化测评平台，实现：
- ✅ 测评用例自动化执行率 ≥ 80%
- ✅ 测评结果自动判定准确率 ≥ 95%
- ✅ 测评报告自动生成时间 ≤ 5 分钟
- ✅ 支持二级、三级系统测评
- ✅ 测评数据自动记录与追踪

### 1.2 建设原则

| 原则 | 说明 |
|------|------|
| 标准遵循 | 严格遵循等保 2.0 标准体系 |
| 安全优先 | 测评过程不影响被测系统安全 |
| 可扩展 | 支持新增测评项和检测脚本 |
| 可追溯 | 测评过程全程留痕、可审计 |

### 4.2 核心脚本示例

#### 4.2.1 密码策略检查脚本

```python
#!/usr/bin/env python3
# -*- coding: utf-8 -*-
"""
密码策略检查脚本
对应测评项：JC-JS-01 身份鉴别 - 密码复杂度检查
依据标准：GB/T 22239-2019 8.1.4.3
"""

import subprocess
import re
from typing import Dict, List
from common.logger import get_logger
from common.database import save_result

logger = get_logger(__name__)

class PasswordPolicyChecker:
    """密码策略检查器"""
    
    # 等保三级密码要求
    REQUIREMENTS = {
        'min_length': 8,           # 最小长度
        'complexity': True,        # 复杂度要求（大小写 + 数字 + 特殊字符）
        'max_age_days': 90,        # 密码最长使用期限
        'history_count': 5,        # 密码历史记忆次数
        'lockout_threshold': 5,    # 登录失败锁定阈值
    }
    
    def check_linux(self, host: str) -> Dict:
        """检查 Linux 系统密码策略"""
        result = {
            'host': host,
            'os_type': 'Linux',
            'items': [],
            'compliant': True,
            'issues': []
        }
        
        # 检查密码最小长度
        try:
            output = subprocess.run(
                ['grep', '^PASS_MIN_LEN', '/etc/login.defs'],
                capture_output=True, text=True, timeout=10
            )
            match = re.search(r'\d+', output.stdout)
            if match:
                min_len = int(match.group())
                passed = min_len >= self.REQUIREMENTS['min_length']
                result['items'].append({
                    'check': '密码最小长度',
                    'expected': f'>= {self.REQUIREMENTS["min_length"]}',
                    'actual': str(min_len),
                    'passed': passed
                })
                if not passed:
                    result['compliant'] = False
                    result['issues'].append(f'密码最小长度 {min_len} 不满足要求')
        except Exception as e:
            logger.error(f'检查密码长度失败：{e}')
        
        # 检查密码复杂度（pam_pwquality）
        # ... 类似逻辑
        
        return result
    
    def check_windows(self, host: str) -> Dict:
        """检查 Windows 系统密码策略"""
        result = {
            'host': host,
            'os_type': 'Windows',
            'items': [],
            'compliant': True,
            'issues': []
        }
        
        # 使用 net accounts 命令
        try:
            output = subprocess.run(
                ['net', 'accounts'],
                capture_output=True, text=True, timeout=10,
                shell=True
            )
            # 解析输出...
        except Exception as e:
            logger.error(f'检查 Windows 密码策略失败：{e}')
        
        return result
    
    def save_to_feishu(self, result: Dict):
        """保存结果到飞书多维表格"""
        # 调用飞书 API 保存结果
        pass


def main():
    """主函数"""
    checker = PasswordPolicyChecker()
    
    # 检查本地系统
    result = checker.check_linux('localhost')
    
    # 保存结果
    save_result('JC-JS-01', result)
    
    # 输出报告
    print(f"检查完成，合规状态：{'符合' if result['compliant'] else '不符合'}")
    for issue in result['issues']:
        print(f"  - {issue}")


if __name__ == '__main__':
    main()
```
